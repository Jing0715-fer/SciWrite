    65→  const [writeOpen, setWriteOpen] = React.useState(false);

    69→  const [gatherOpen, setGatherOpen] = React.useState(false);
    70→  const [insightsOpen, setInsightsOpen] = React.useState(false);
    71→  const [paletteOpen, setPaletteOpen] = React.useState(false);
    72→  const [outlineOpen, setOutlineOpen] = React.useState(false);
    73→  const [userDataOpen, setUserDataOpen] = React.useState(false);

    75→  const [llmConfigOpen, setLlmConfigOpen] = React.useState(false);
    76→

    87→
    88→  // listen for project-created event
    89→  React.useEffect(() => {
    90→    const handler = (e: Event) => {
    91→      const id = (e as CustomEvent).detail as string;
    92→      setActiveProjectId(id);
    93→    };
    94→    window.addEventListener("sciwrite:select-project", handler);
    95→    return () => window.removeEventListener("sciwrite:select-project", handler);
    96→  }, []);
    97→
    98→  // auto-select first project
    99→  React.useEffect(() => {
   100→    if (!activeProjectId && projectsQ.data?.projects.length) {
   101→      setActiveProjectId(projectsQ.data.projects[0].id);
   102→    }

   104→
   105→  const projects = projectsQ.data?.projects ?? [];
   106→  const project = projectQ.data?.project;
   107→  const paragraphs = (project?.paragraphs ?? []) as any[];
   108→  const dataSources = project?.dataSources ?? [];
   109→  const articles = (project?.articles ?? []) as any[];
   110→  const references = React.useMemo(() => {
   111→    const map = new Map<string, any>();
   112→    for (const r of project?.references ?? []) {
   113→      const key = `${r.type}:${r.externalId || r.title}`;
   114→      if (!map.has(key)) map.set(key, r);
   115→    }
   116→    for (const p of paragraphs) {
   117→      for (const r of p.references || []) {
   118→        const key = `${r.type}:${r.externalId || r.title}`;
   119→        if (!map.has(key)) map.set(key, r);
   120→      }
   121→    }
   122→    return [...map.values()];
   123→  }, [paragraphs, project?.references]);
   124→
   125→  // Derived progress stats
   126→  const progressStats = React.useMemo(() => {
   127→    const totalWords = paragraphs.reduce(
   128→      (sum, p) => sum + (p.wordCount || 0),
   129→      0
   130→    );
   131→    const totalCitations = paragraphs.reduce((sum, p) => {
   132→      const matches = p.content?.match(
   133→        /\[(\d{1,3}(?:[,\-–\s]\d{1,3})*|[A-Z]{2,12}:\s?[^\]\n]{1,60})\]/g
   134→      );
   135→      return sum + (matches?.length || 0);
   136→    }, 0);
   137→    const paragraphsCited = paragraphs.filter(
   138→      (p) => /\[\d{1,3}/.test(p.content || "") || /\[[A-Z]{2,12}:/i.test(p.content || "")
   139→    ).length;
   140→    const citationCoverage =
   141→      paragraphs.length > 0
   142→        ? Math.round((paragraphsCited / paragraphs.length) * 100)
   143→        : 0;
   144→    const unresolved = paragraphs.reduce(
   145→      (s, p) => s + (p.annotations?.filter((a: any) => !a.resolved).length || 0),
   146→      0
   147→    );
   148→    const resolved = paragraphs.reduce(
   149→      (s, p) => s + (p.annotations?.filter((a: any) => a.resolved).length || 0),
   150→      0
   151→    );
   152→    return {
   153→      totalWords,
   154→      totalParagraphs: paragraphs.length,
   155→      totalCitations,
   156→      citationCoverage,
   157→      unresolvedAnnotations: unresolved,
   158→      resolvedAnnotations: resolved,
   159→    };
   160→  }, [paragraphs]);
   161→
   162→  const [wordGoal, setWordGoal] = React.useState(1000);
   163→
   164→  // Keyboard shortcuts (defined after paragraphs so it can reference it)
   165→  React.useEffect(() => {
   166→    const handler = (e: KeyboardEvent) => {
   167→      const meta = e.metaKey || e.ctrlKey;
   168→      const tag = (e.target as HTMLElement)?.tagName;
   169→      const isTyping =
   170→        tag === "INPUT" || tag === "TEXTAREA" || (e.target as HTMLElement)?.isContentEditable;
   171→      // Cmd/Ctrl+K always opens palette
   172→      if (meta && e.key.toLowerCase() === "k") {
   173→        e.preventDefault();
   174→        setPaletteOpen((v) => !v);
   175→        return;
   176→      }
   177→      // Single-key shortcuts only when not typing and no modifier (except Shift)
   178→      if (isTyping || meta || e.altKey) return;
   179→      if (!activeProjectId) return;
   180→      const k = e.key.toLowerCase();
   181→      if (k === "n") {
   182→        e.preventDefault();
   183→        setWriteOpen(true);
   184→      } else if (k === "g") {
   185→        e.preventDefault();
   186→        setGatherOpen(true);
   187→      } else if (k === "i") {
   188→        e.preventDefault();
   189→        setInsightsOpen(true);
   190→      } else if (k === "o") {
   191→        e.preventDefault();
   192→        setOutlineOpen(true);
   193→      } else if (k === "c" && paragraphs.length >= 2) {
   194→        e.preventDefault();
   195→        setComposeOpen(true);
   196→      } else if (k === "d") {
   197→        e.preventDefault();
   198→        const isDark = document.documentElement.classList.contains("dark");
   199→        document.documentElement.classList.toggle("dark", !isDark);
   200→        try {
   201→          localStorage.setItem("theme", isDark ? "light" : "dark");
   202→        } catch {}
   203→      }
   204→    };
   205→    window.addEventListener("keydown", handler);
   206→    return () => window.removeEventListener("keydown", handler);
   207→  }, [activeProjectId, paragraphs.length]);
   208→
   209→  return (
   210→    <div className="h-screen flex flex-col bg-background overflow-hidden">
   211→      <Header
   212→        project={project}
   213→        onOpenWrite={() => setWriteOpen(true)}
   214→        onOpenCompose={() => setComposeOpen(true)}
   215→        onOpenGather={() => setGatherOpen(true)}
   216→        onOpenInsights={() => setInsightsOpen(true)}
   217→        onOpenOutline={() => setOutlineOpen(true)}
   218→        onOpenOneClick={() => setOneClickOpen(true)}
   219→        onOpenLLMConfig={() => setLlmConfigOpen(true)}
   220→        paragraphCount={paragraphs.length}
   221→        articleCount={articles.length}
   222→      />
   223→
   224→      <main className="flex-1 min-h-0 px-3 pb-2">
   225→        <ResizablePanelGroup direction="horizontal" className="rounded-xl border border-border/60 bg-card overflow-hidden h-full">
   226→          {/* Left: projects */}
   227→          <ResizablePanel defaultSize={18} minSize={15} maxSize={28} className="bg-sidebar/50">
   228→            <ProjectsSidebar
   229→              projects={projects}
   230→              activeId={activeProjectId}
   231→              onSelect={setActiveProjectId}
   232→            />
   233→          </ResizablePanel>
   234→          <ResizableHandle withHandle />
   235→
   236→          {/* Center: writing workspace */}
   237→          <ResizablePanel defaultSize={52} minSize={35} className="min-w-0">
   238→            <WritingWorkspace
   239→              project={project}
   240→              paragraphs={paragraphs}
   241→              articles={articles}
   242→              references={references}
   243→              activeProjectId={activeProjectId}
   244→              onOpenWrite={() => setWriteOpen(true)}
   245→              onOpenCompose={() => setComposeOpen(true)}
   246→              onOpenGather={() => setGatherOpen(true)}
   247→              onOpenOutline={() => setOutlineOpen(true)}
   248→              progressStats={progressStats}
   249→              wordGoal={wordGoal}
   250→              onWordGoalChange={setWordGoal}
   251→              tipsOpen={tipsOpen}
   252→              onTipsOpenChange={setTipsOpen}
   253→              onOpenUserData={() => setUserDataOpen(true)}
   254→              onOpenArticle={(a) => setViewArticle(a as Article)}
   255→            />
   256→          </ResizablePanel>
   257→          <ResizableHandle withHandle />
   258→
   259→          {/* Right: databases + knowledge */}
   260→          <ResizablePanel defaultSize={30} minSize={24} maxSize={42} className="bg-sidebar/30">
   261→            <div className="flex flex-col h-full overflow-hidden">
   262→              <div className="h-[44%] min-h-0 border-b border-border/60 overflow-hidden">
   263→                <DatabaseQueryPanel projectId={activeProjectId} />
   264→              </div>
   265→              <div className="flex-1 min-h-0 overflow-hidden">
   266→                <KnowledgePanel
   267→                  projectId={activeProjectId}
   268→                  dataSources={dataSources}
   269→                  references={references}
   270→                  articles={articles}
   271→                  onOpenArticle={(a) => setViewArticle(a as Article)}
   272→                />
   273→              </div>
   274→            </div>
   275→          </ResizablePanel>
   276→        </ResizablePanelGroup>
   277→      </main>
   278→
   279→      <Footer onOpenPalette={() => setPaletteOpen(true)} />
   280→
   281→      {/* Modals */}
   282→      {activeProjectId && project && (
   283→        <TopicComposer
   284→          open={writeOpen}
   285→          onOpenChange={setWriteOpen}
   286→          projectId={activeProjectId}
   287→          projectTopic={project.topic}
   288→          projectField={project.field}
   289→          references={references.map((r) => ({
   290→            id: r.id,
   291→            title: r.title,
   292→            type: r.type,
   293→            externalId: r.externalId,
   294→          }))}
   295→          dataSources={dataSources.map((d) => ({
   296→            id: d.id,
   297→            source: d.source,
   298→            title: d.title,
   299→            query: d.query,
   300→          }))}
   301→        />
   302→      )}
   303→      {activeProjectId && (
   304→        <ArticleComposer
   305→          open={composeOpen}
   306→          onOpenChange={setComposeOpen}
   307→          projectId={activeProjectId}
   308→          paragraphs={paragraphs}
   309→        />
   310→      )}
   311→      {viewArticle && (
   312→        <ArticleViewerWithTabs
   313→          article={viewArticle}
   314→          projectId={activeProjectId!}
   315→          onClose={() => setViewArticle(null)}
   316→        />
   317→      )}
   318→      {activeProjectId && project && (
   319→        <DataGatheringDialog
   320→          open={gatherOpen}
   321→          onOpenChange={setGatherOpen}
   322→          projectId={activeProjectId}
   323→          topic={project.topic}
   324→          field={project.field}
   325→          onProceedToWrite={() => setWriteOpen(true)}
   326→        />
   327→      )}
   328→      {activeProjectId && (
   329→        <InsightsDialog
   330→          open={insightsOpen}
   331→          onOpenChange={setInsightsOpen}
   332→          projectId={activeProjectId}
   333→        />
   334→      )}
   335→      {activeProjectId && project && (
   336→        <OutlineDialog
   337→          open={outlineOpen}
   338→          onOpenChange={setOutlineOpen}
   339→          projectId={activeProjectId}
   340→          topic={project.topic}
   341→          onUseParagraph={() => {
   342→            setOutlineOpen(false);
   343→            setWriteOpen(true);
   344→          }}
   345→        />
   346→      )}
   347→      {activeProjectId && (
   348→        <UserDataDialog
   349→          open={userDataOpen}
   350→          onOpenChange={setUserDataOpen}
   351→          projectId={activeProjectId}
   352→        />
   353→      )}
   354→      {activeProjectId && project && (
   355→        <OneClickGenerateDialog
   356→          open={oneClickOpen}
   357→          onOpenChange={setOneClickOpen}
   358→          projectId={activeProjectId}
   359→          topic={project.topic}
   360→        />
   361→      )}
   362→      <LLMConfigDialog
   363→        open={llmConfigOpen}
   364→        onOpenChange={setLlmConfigOpen}
   365→      />
   366→      <CommandPalette
   367→        open={paletteOpen}
   368→        onOpenChange={setPaletteOpen}
   369→        actions={[
   370→          {
   371→            id: "write",
   372→            label: "AI Write paragraph",
   373→            hint: "Draft a citation-backed paragraph",
   374→            icon: <Sparkles className="h-3.5 w-3.5" />,
   375→            shortcut: "N",
   376→            onSelect: () => setWriteOpen(true),
   377→            group: "Writing",
   378→            disabled: !activeProjectId,
   379→          },
   380→          {
   381→            id: "gather",
   382→            label: "Gather sources (AI)",
   383→            hint: "Clarify → organize → adversarial critique",
   384→            icon: <Radar className="h-3.5 w-3.5" />,
   385→            shortcut: "G",
   386→            onSelect: () => setGatherOpen(true),
   387→            group: "Writing",
   388→            disabled: !activeProjectId,
   389→          },
   390→          {
   391→            id: "compose",
   392→            label: "Compose article",
   393→            hint: "Stitch paragraphs into a deeper article",
   394→            icon: <Layers className="h-3.5 w-3.5" />,
   395→            shortcut: "C",
   396→            onSelect: () => setComposeOpen(true),
   397→            group: "Writing",
   398→            disabled: paragraphs.length < 2,
   399→          },
   400→          {
   401→            id: "insights",
   402→            label: "Project insights",
   403→            hint: "Word count, citation coverage, distributions",
   404→            icon: <BarChart3 className="h-3.5 w-3.5" />,
   405→            shortcut: "I",
   406→            onSelect: () => setInsightsOpen(true),
   407→            group: "Project",
   408→            disabled: !activeProjectId,
   409→          },
   410→          {
   411→            id: "outline",
   412→            label: "Generate research outline",
   413→            hint: "AI-suggested paragraph plan with queries",
   414→            icon: <ListTree className="h-3.5 w-3.5" />,
   415→            shortcut: "O",
   416→            onSelect: () => setOutlineOpen(true),
   417→            group: "Writing",
   418→            disabled: !activeProjectId,
   419→          },
   420→          {
   421→            id: "dark",
   422→            label: "Toggle dark mode",
   423→            icon: <Moon className="h-3.5 w-3.5" />,
   424→            shortcut: "D",
   425→            onSelect: () => {
   426→              const isDark = document.documentElement.classList.contains("dark");
   427→              document.documentElement.classList.toggle("dark", !isDark);
   428→              try {
   429→                localStorage.setItem("theme", isDark ? "light" : "dark");
   430→              } catch {}
   431→            },
   432→            group: "Project",
   433→          },
   434→        ]}
   435→      />
   436→    </div>

   438→}

   520→              variant="ghost"
   521→              size="sm"
   522→              className="h-8 text-xs gap-1.5"

   781→function EmbeddedReviewWorkspace({ articleId, articleTitle, projectId }: { articleId?: string; articleTitle?: string; projectId: string }) {
   782→  const qc = useQueryClient();
   783→  // Load saved review from DB on mount
   784→  const { data: savedReview } = useQuery({
   785→    queryKey: ["saved-review", articleId],
   786→    queryFn: () => articleId ? api.getSavedReview(articleId) : Promise.resolve({ notFound: true }),
   787→    enabled: !!articleId,
   788→    staleTime: Infinity,
   789→  });
   790→  const { data: reviewData } = useQuery({
   791→    queryKey: ["article-review", articleId],
   792→    queryFn: () => api.aiReview({ mode: "review", articleId: articleId! }),
   793→    enabled: false,
   794→    staleTime: Infinity,
   795→  });
   796→
   797→  const reviewMut = useMutation({
   798→    mutationFn: () => articleId ? api.aiReview({ mode: "review", articleId }) : Promise.reject(new Error("No article")),
   799→    onSuccess: (data) => {
   800→      qc.setQueryData(["article-review", articleId], data);
   801→      qc.invalidateQueries({ queryKey: ["saved-review", articleId] });
   802→      toast.success("Review completed.");
   803→    },
   804→    onError: (e: Error) => toast.error(e.message),
   805→  });
   806→
   807→  // Use freshly-run review if available, else saved from DB
   808→  const displayData = reviewData || (savedReview && !savedReview.notFound ? savedReview : null);
   809→
   810→  return (
   811→    <ScrollArea className="flex-1 min-h-0 scroll-academic">
   812→      <div className="px-5 py-4 max-w-2xl mx-auto">
   813→        {!displayData && !reviewMut.isPending && (
   814→          <div className="text-center py-12">
   815→            <div className="h-14 w-14 mx-auto rounded-2xl bg-primary/10 flex items-center justify-center mb-3">

   820→              Run a structured peer review with multi-dimensional scoring.
   821→            </p>
   822→            <Button size="sm" className="gap-1.5 text-xs" onClick={() => reviewMut.mutate()} disabled={!articleId}>
   823→              <Gavel className="h-3.5 w-3.5" /> Run review
   824→            </Button>
   825→          </div>
   826→        )}

   828→          <div className="flex items-center justify-center py-12"><Loader2 className="h-6 w-6 animate-spin text-primary" /></div>

   830→        {displayData && !reviewMut.isPending && (
   831→          <div className="space-y-3">
   832→            {displayData.verdict && (
   833→              <div className={`rounded-lg border p-3 ${displayData.verdict === "accept" ? "border-emerald-200/60 bg-emerald-50/50" : "border-amber-200/60 bg-amber-50/50"}`}>
   834→                <span className={`text-sm font-semibold ${displayData.verdict === "accept" ? "text-emerald-700" : "text-amber-700"}`}>
   835→                  {displayData.verdict === "accept" ? "✓ Accept" : `⚠ ${displayData.verdict}`}
   836→                </span>

   878→function RelationshipWorkspace({ projectId }: { projectId: string }) {
   879→  const qc = useQueryClient();
   880→  // First try to load saved analysis from DB
   881→  const { data: savedRel } = useQuery({
   882→    queryKey: ["saved-relationships", projectId],
   883→    queryFn: () => api.getSavedRelationships(projectId),
   884→    enabled: !!projectId,
   885→    staleTime: Infinity,
   886→  });
   887→  const { data: freshRel, isLoading: loading, error: relError } = useQuery({
   888→    queryKey: ["source-relationships", projectId],
   889→    queryFn: async () => {
   890→      const res = await fetch("/api/ai/source-relationships", {
   891→        method: "POST",
   892→        headers: { "Content-Type": "application/json" },
   893→        body: JSON.stringify({ projectId }),
   894→      });
   895→      if (!res.ok) {
   896→        const text = await res.text();
   897→        throw new Error(text.slice(0, 200) || `Analysis failed (${res.status})`);
   898→      }
   899→      const data = await res.json();
   900→      qc.invalidateQueries({ queryKey: ["saved-relationships", projectId] });
   901→      return data;
   902→    },
   903→    enabled: false, // Only on manual trigger
   904→  });
   905→  const relMut = useMutation({
   906→    mutationFn: () =>
   907→      fetch("/api/ai/source-relationships", {
   908→        method: "POST",
   909→        headers: { "Content-Type": "application/json" },
   910→        body: JSON.stringify({ projectId }),
   911→      }).then(async (res) => {
   912→        if (!res.ok) {
   913→          const text = await res.text();
   914→          throw new Error(text.slice(0, 200) || `Analysis failed (${res.status})`);
   915→        }
   916→        return res.json();
   917→      }),
   918→    onSuccess: () => {
   919→      qc.invalidateQueries({ queryKey: ["saved-relationships", projectId] });
   920→      toast.success("Relationship analysis complete.");

   922→    onError: (e: Error) => toast.error(e.message),

   925→  const relData = freshRel || (savedRel && !savedRel.notFound ? savedRel : null);
   926→
   927→  return (
   928→    <ScrollArea className="flex-1 min-h-0 scroll-academic">
   929→      <div className="px-5 py-4 max-w-2xl mx-auto">

   940→              <Network className="h-3.5 w-3.5" /> Retry
   941→            </Button>
   942→          </div>
   943→        )}
   944→        {!relMut.isPending && !relError && relData?.skipped && (
   945→          <div className="text-center py-12">
   946→            <Network className="h-10 w-10 mx-auto opacity-40 mb-3" />
