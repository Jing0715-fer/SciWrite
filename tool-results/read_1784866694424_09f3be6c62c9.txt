  1378→export type TranslationKey = keyof typeof translations.en;
  1379→
  1380→interface I18nContextValue {
  1381→  lang: Lang;
  1382→  setLang: (l: Lang) => void;
  1383→  t: (key: TranslationKey, vars?: Record<string, string | number>) => string;
  1384→}
  1385→
  1386→const I18nContext = React.createContext<I18nContextValue>({
  1387→  lang: "en",
  1388→  setLang: () => {},
  1389→  t: (key) => key,
  1390→});
  1391→
  1392→export function I18nProvider({ children }: { children: React.ReactNode }) {
  1393→  const [lang, setLangState] = React.useState<Lang>("en");
  1394→
  1395→  React.useEffect(() => {
  1396→    try {
  1397→      const saved = localStorage.getItem("sciwrite-lang") as Lang | null;
  1398→      if (saved === "en" || saved === "zh") {
  1399→        setLangState(saved);
  1400→      }
  1401→    } catch {}
  1402→  }, []);
  1403→
  1404→  const setLang = React.useCallback((l: Lang) => {
  1405→    setLangState(l);
  1406→    try {
  1407→      localStorage.setItem("sciwrite-lang", l);
  1408→    } catch {}
  1409→  }, []);
  1410→
  1411→  const t = React.useCallback(
  1412→    (key: TranslationKey, vars?: Record<string, string | number>) => {
  1413→      const raw = (translations[lang] as Record<string, string>)[key] || key;
  1414→      if (!vars) return raw;
  1415→      return raw.replace(/\{(\w+)\}/g, (_, name: string) =>
  1416→        vars[name] !== undefined ? String(vars[name]) : `{${name}}`
  1417→      );
  1418→    },
  1419→    [lang]
  1420→  );
  1421→
  1422→  return (
  1423→    <I18nContext.Provider value={{ lang, setLang, t }}>
  1424→      {children}
  1425→    </I18nContext.Provider>
  1426→  );
  1427→}
  1428→
  1429→export function useI18n() {
  1430→  return React.useContext(I18nContext);
  1431→}
