    66→  const [writeOpen, setWriteOpen] = React.useState(false);

    69→  const [viewArticle, setViewArticle] = React.useState<Article | null>(null);
    70→  const [gatherOpen, setGatherOpen] = React.useState(false);
    71→  const [insightsOpen, setInsightsOpen] = React.useState(false);
    72→  const [paletteOpen, setPaletteOpen] = React.useState(false);
    73→  const [outlineOpen, setOutlineOpen] = React.useState(false);
    74→  const [userDataOpen, setUserDataOpen] = React.useState(false);
    75→  const [oneClickOpen, setOneClickOpen] = React.useState(false);
    76→  const [llmConfigOpen, setLlmConfigOpen] = React.useState(false);
    77→
    78→  const projectsQ = useQuery({

    82→

    88→
    89→  // listen for project-created event
    90→  React.useEffect(() => {
    91→    const handler = (e: Event) => {
    92→      const id = (e as CustomEvent).detail as string;
    93→      setActiveProjectId(id);
    94→    };

    96→    return () => window.removeEventListener("sciwrite:select-project", handler);
    97→  }, []);
    98→
    99→  // auto-select first project
   100→  React.useEffect(() => {
   101→    if (!activeProjectId && projectsQ.data?.projects.length) {
   102→      setActiveProjectId(projectsQ.data.projects[0].id);
   103→    }

   105→
   106→  const projects = projectsQ.data?.projects ?? [];
   107→  const project = projectQ.data?.project;
   108→  const paragraphs = (project?.paragraphs ?? []) as any[];
   109→  const dataSources = project?.dataSources ?? [];
   110→  const articles = (project?.articles ?? []) as any[];
   111→  const references = React.useMemo(() => {
   112→    const map = new Map<string, any>();
   113→    for (const r of project?.references ?? []) {
   114→      const key = `${r.type}:${r.externalId || r.title}`;
   115→      if (!map.has(key)) map.set(key, r);
   116→    }
   117→    for (const p of paragraphs) {
   118→      for (const r of p.references || []) {
   119→        const key = `${r.type}:${r.externalId || r.title}`;
   120→        if (!map.has(key)) map.set(key, r);
   121→      }
   122→    }
   123→    return [...map.values()];
   124→  }, [paragraphs, project?.references]);
   125→
   126→  // Derived progress stats
   127→  const progressStats = React.useMemo(() => {
   128→    const totalWords = paragraphs.reduce(
   129→      (sum, p) => sum + (p.wordCount || 0),
   130→      0
   131→    );
   132→    const totalCitations = paragraphs.reduce((sum, p) => {
   133→      const matches = p.content?.match(
   134→        /\[(\d{1,3}(?:[,\-–\s]\d{1,3})*|[A-Z]{2,12}:\s?[^\]\n]{1,60})\]/g
   135→      );
   136→      return sum + (matches?.length || 0);
   137→    }, 0);
   138→    const paragraphsCited = paragraphs.filter(
   139→      (p) => /\[\d{1,3}/.test(p.content || "") || /\[[A-Z]{2,12}:/i.test(p.content || "")
   140→    ).length;
   141→    const citationCoverage =
   142→      paragraphs.length > 0
   143→        ? Math.round((paragraphsCited / paragraphs.length) * 100)
   144→        : 0;
   145→    const unresolved = paragraphs.reduce(
   146→      (s, p) => s + (p.annotations?.filter((a: any) => !a.resolved).length || 0),
   147→      0
   148→    );
   149→    const resolved = paragraphs.reduce(
   150→      (s, p) => s + (p.annotations?.filter((a: any) => a.resolved).length || 0),
   151→      0
   152→    );
   153→    return {
   154→      totalWords,
   155→      totalParagraphs: paragraphs.length,
   156→      totalCitations,
   157→      citationCoverage,
   158→      unresolvedAnnotations: unresolved,
   159→      resolvedAnnotations: resolved,
   160→    };
   161→  }, [paragraphs]);
   162→
   163→  const [wordGoal, setWordGoal] = React.useState(1000);
   164→
   165→  // Keyboard shortcuts (defined after paragraphs so it can reference it)
   166→  React.useEffect(() => {
   167→    const handler = (e: KeyboardEvent) => {
   168→      const meta = e.metaKey || e.ctrlKey;
   169→      const tag = (e.target as HTMLElement)?.tagName;
   170→      const isTyping =
   171→        tag === "INPUT" || tag === "TEXTAREA" || (e.target as HTMLElement)?.isContentEditable;
   172→      // Cmd/Ctrl+K always opens palette
   173→      if (meta && e.key.toLowerCase() === "k") {
   174→        e.preventDefault();
   175→        setPaletteOpen((v) => !v);
   176→        return;
   177→      }
   178→      // Single-key shortcuts only when not typing and no modifier (except Shift)
   179→      if (isTyping || meta || e.altKey) return;
   180→      if (!activeProjectId) return;
   181→      const k = e.key.toLowerCase();
   182→      if (k === "n") {
   183→        e.preventDefault();
   184→        setWriteOpen(true);
   185→      } else if (k === "g") {
   186→        e.preventDefault();
   187→        setGatherOpen(true);
   188→      } else if (k === "i") {
   189→        e.preventDefault();
   190→        setInsightsOpen(true);
   191→      } else if (k === "o") {
   192→        e.preventDefault();
   193→        setOutlineOpen(true);
   194→      } else if (k === "c" && paragraphs.length >= 2) {
   195→        e.preventDefault();
   196→        setComposeOpen(true);
   197→      } else if (k === "d") {
   198→        e.preventDefault();
   199→        const isDark = document.documentElement.classList.contains("dark");
   200→        document.documentElement.classList.toggle("dark", !isDark);
   201→        try {
   202→          localStorage.setItem("theme", isDark ? "light" : "dark");
   203→        } catch {}
   204→      }
   205→    };
   206→    window.addEventListener("keydown", handler);
   207→    return () => window.removeEventListener("keydown", handler);
   208→  }, [activeProjectId, paragraphs.length]);
   209→
   210→  return (
   211→    <div className="h-screen flex flex-col bg-background overflow-hidden">
   212→      <Header
   213→        project={project}
   214→        onOpenWrite={() => setWriteOpen(true)}
   215→        onOpenCompose={() => setComposeOpen(true)}
   216→        onOpenGather={() => setGatherOpen(true)}
   217→        onOpenInsights={() => setInsightsOpen(true)}
   218→        onOpenOutline={() => setOutlineOpen(true)}
   219→        onOpenOneClick={() => setOneClickOpen(true)}
   220→        onOpenLLMConfig={() => setLlmConfigOpen(true)}
   221→        paragraphCount={paragraphs.length}
   222→        articleCount={articles.length}
   223→      />
   224→
   225→      <main className="flex-1 min-h-0 px-3 pb-2">
   226→        <ResizablePanelGroup direction="horizontal" className="rounded-xl border border-border/60 bg-card overflow-hidden h-full">
   227→          {/* Left: projects */}
   228→          <ResizablePanel defaultSize={18} minSize={15} maxSize={28} className="bg-sidebar/50">
   229→            <ProjectsSidebar
   230→              projects={projects}
   231→              activeId={activeProjectId}
   232→              onSelect={setActiveProjectId}
   233→            />
   234→          </ResizablePanel>
   235→          <ResizableHandle withHandle />
   236→
   237→          {/* Center: writing workspace */}
   238→          <ResizablePanel defaultSize={52} minSize={35} className="min-w-0">
   239→            <WritingWorkspace
   240→              project={project}
   241→              paragraphs={paragraphs}
   242→              articles={articles}
   243→              references={references}
   244→              activeProjectId={activeProjectId}
   245→              onOpenWrite={() => setWriteOpen(true)}
   246→              onOpenCompose={() => setComposeOpen(true)}
   247→              onOpenGather={() => setGatherOpen(true)}
   248→              onOpenOutline={() => setOutlineOpen(true)}
   249→              progressStats={progressStats}
   250→              wordGoal={wordGoal}
   251→              onWordGoalChange={setWordGoal}
   252→              tipsOpen={tipsOpen}
   253→              onTipsOpenChange={setTipsOpen}
   254→              onOpenUserData={() => setUserDataOpen(true)}
   255→              onOpenArticle={(a) => setViewArticle(a as Article)}
   256→            />
   257→          </ResizablePanel>
   258→          <ResizableHandle withHandle />
   259→
   260→          {/* Right: databases + knowledge */}
   261→          <ResizablePanel defaultSize={30} minSize={24} maxSize={42} className="bg-sidebar/30">
   262→            <div className="flex flex-col h-full overflow-hidden">
   263→              <div className="h-[44%] min-h-0 border-b border-border/60 overflow-hidden">
   264→                <DatabaseQueryPanel projectId={activeProjectId} />
   265→              </div>
   266→              <div className="flex-1 min-h-0 overflow-hidden">
   267→                <KnowledgePanel
   268→                  projectId={activeProjectId}
   269→                  dataSources={dataSources}
   270→                  references={references}
   271→                  articles={articles}
   272→                  onOpenArticle={(a) => setViewArticle(a as Article)}
   273→                />
   274→              </div>
   275→            </div>
   276→          </ResizablePanel>
   277→        </ResizablePanelGroup>
   278→      </main>
   279→
   280→      <Footer onOpenPalette={() => setPaletteOpen(true)} />
   281→
   282→      {/* Modals */}
   283→      {activeProjectId && project && (
   284→        <TopicComposer
   285→          open={writeOpen}
   286→          onOpenChange={setWriteOpen}
   287→          projectId={activeProjectId}
   288→          projectTopic={project.topic}
   289→          projectField={project.field}
   290→          references={references.map((r) => ({
   291→            id: r.id,
   292→            title: r.title,
   293→            type: r.type,
   294→            externalId: r.externalId,
   295→          }))}
   296→          dataSources={dataSources.map((d) => ({
   297→            id: d.id,
   298→            source: d.source,
   299→            title: d.title,
   300→            query: d.query,
   301→          }))}
   302→        />
   303→      )}
   304→      {activeProjectId && (
   305→        <ArticleComposer
   306→          open={composeOpen}
   307→          onOpenChange={setComposeOpen}
   308→          projectId={activeProjectId}
   309→          paragraphs={paragraphs}
   310→        />
   311→      )}
   312→      {viewArticle && (
   313→        <ArticleViewerWithTabs
   314→          article={viewArticle}
   315→          projectId={activeProjectId!}

   317→        />
   318→      )}
   319→      {activeProjectId && project && (
   320→        <DataGatheringDialog
   321→          open={gatherOpen}
   322→          onOpenChange={setGatherOpen}
   323→          projectId={activeProjectId}
   324→          topic={project.topic}
   325→          field={project.field}
   326→          onProceedToWrite={() => setWriteOpen(true)}
   327→        />
   328→      )}
   329→      {activeProjectId && (
   330→        <InsightsDialog
   331→          open={insightsOpen}
   332→          onOpenChange={setInsightsOpen}
   333→          projectId={activeProjectId}
   334→        />
   335→      )}
   336→      {activeProjectId && project && (
   337→        <OutlineDialog
   338→          open={outlineOpen}
   339→          onOpenChange={setOutlineOpen}
   340→          projectId={activeProjectId}
   341→          topic={project.topic}
   342→          onUseParagraph={() => {
   343→            setOutlineOpen(false);
   344→            setWriteOpen(true);
   345→          }}
   346→        />
   347→      )}
   348→      {activeProjectId && (
   349→        <UserDataDialog
   350→          open={userDataOpen}
   351→          onOpenChange={setUserDataOpen}
   352→          projectId={activeProjectId}
   353→        />
   354→      )}
   355→      {activeProjectId && project && (
   356→        <OneClickGenerateDialog
   357→          open={oneClickOpen}
   358→          onOpenChange={setOneClickOpen}
   359→          projectId={activeProjectId}
   360→          topic={project.topic}
   361→        />
   362→      )}
   363→      <LLMConfigDialog
   364→        open={llmConfigOpen}
   365→        onOpenChange={setLlmConfigOpen}
   366→      />
   367→      <CommandPalette
   368→        open={paletteOpen}
   369→        onOpenChange={setPaletteOpen}
   370→        actions={[
   371→          {
   372→            id: "write",
   373→            label: t("cmd.writeParagraph"),
   374→            hint: t("cmd.writeHint"),
   375→            icon: <Sparkles className="h-3.5 w-3.5" />,
   376→            shortcut: "N",
   377→            onSelect: () => setWriteOpen(true),
   378→            group: t("cmd.groupWriting"),
   379→            disabled: !activeProjectId,
   380→          },
   381→          {
   382→            id: "gather",
   383→            label: t("cmd.gatherSourcesAction"),
   384→            hint: t("cmd.gatherDesc"),
   385→            icon: <Radar className="h-3.5 w-3.5" />,
   386→            shortcut: "G",
   387→            onSelect: () => setGatherOpen(true),
   388→            group: t("cmd.groupWriting"),
   389→            disabled: !activeProjectId,
   390→          },
   391→          {
   392→            id: "compose",
   393→            label: t("cmd.composeArticle"),
   394→            hint: t("cmd.composeHint"),
   395→            icon: <Layers className="h-3.5 w-3.5" />,
   396→            shortcut: "C",
   397→            onSelect: () => setComposeOpen(true),
   398→            group: t("cmd.groupWriting"),
   399→            disabled: paragraphs.length < 2,
   400→          },
   401→          {
   402→            id: "insights",
   403→            label: t("cmd.projectInsights"),
   404→            hint: t("cmd.insightsHint"),
   405→            icon: <BarChart3 className="h-3.5 w-3.5" />,
   406→            shortcut: "I",
   407→            onSelect: () => setInsightsOpen(true),
   408→            group: t("cmd.groupProject"),
   409→            disabled: !activeProjectId,
   410→          },
   411→          {
   412→            id: "outline",
   413→            label: t("cmd.generateOutline"),
   414→            hint: t("cmd.outlineHint"),
   415→            icon: <ListTree className="h-3.5 w-3.5" />,
   416→            shortcut: "O",
   417→            onSelect: () => setOutlineOpen(true),
   418→            group: t("cmd.groupWriting"),
   419→            disabled: !activeProjectId,
   420→          },
   421→          {
   422→            id: "dark",
   423→            label: t("cmd.toggleDark"),
   424→            icon: <Moon className="h-3.5 w-3.5" />,
   425→            shortcut: "D",
   426→            onSelect: () => {
   427→              const isDark = document.documentElement.classList.contains("dark");
   428→              document.documentElement.classList.toggle("dark", !isDark);
   429→              try {
   430→                localStorage.setItem("theme", isDark ? "light" : "dark");
   431→              } catch {}
   432→            },
   433→            group: t("cmd.groupProject"),
   434→          },
   435→        ]}
   436→      />
   437→    </div>

   782→function EmbeddedReviewWorkspace({ articleId, articleTitle, projectId }: { articleId?: string; articleTitle?: string; projectId: string }) {
   783→  const qc = useQueryClient();
   784→  // Load saved review from DB on mount
   785→  const { data: savedReview } = useQuery({
   786→    queryKey: ["saved-review", articleId],
   787→    queryFn: () => articleId ? api.getSavedReview(articleId) : Promise.resolve({ notFound: true }),
   788→    enabled: !!articleId,
   789→    staleTime: Infinity,
   790→  });
   791→  const { data: reviewData } = useQuery({
   792→    queryKey: ["article-review", articleId],
   793→    queryFn: () => api.aiReview({ mode: "review", articleId: articleId! }),
   794→    enabled: false,
   795→    staleTime: Infinity,
   796→  });
   797→
   798→  const reviewMut = useMutation({
   799→    mutationFn: () => articleId ? api.aiReview({ mode: "review", articleId }) : Promise.reject(new Error("No article")),
   800→    onSuccess: (data) => {
   801→      qc.setQueryData(["article-review", articleId], data);
   802→      qc.invalidateQueries({ queryKey: ["saved-review", articleId] });
   803→      toast.success("Review completed.");
   804→    },
   805→    onError: (e: Error) => toast.error(e.message),
   806→  });
   807→
   808→  // Use freshly-run review if available, else saved from DB
   809→  const displayData = reviewData || (savedReview && !savedReview.notFound ? savedReview : null);
   810→
   811→  return (
   812→    <ScrollArea className="flex-1 min-h-0 scroll-academic">
   813→      <div className="px-5 py-4 max-w-2xl mx-auto">
   814→        {!displayData && !reviewMut.isPending && (
   815→          <div className="text-center py-12">

   821→              Run a structured peer review with multi-dimensional scoring.
   822→            </p>
   823→            <Button size="sm" className="gap-1.5 text-xs" onClick={() => reviewMut.mutate()} disabled={!articleId}>
   824→              <Gavel className="h-3.5 w-3.5" /> Run review
   825→            </Button>
   826→          </div>
   827→        )}

   829→          <div className="flex items-center justify-center py-12"><Loader2 className="h-6 w-6 animate-spin text-primary" /></div>
   830→        )}
   831→        {displayData && !reviewMut.isPending && (
   832→          <div className="space-y-3">
   833→            {displayData.verdict && (

   835→                <span className={`text-sm font-semibold ${displayData.verdict === "accept" ? "text-emerald-700" : "text-amber-700"}`}>
   836→                  {displayData.verdict === "accept" ? "✓ Accept" : `⚠ ${displayData.verdict}`}

   879→function RelationshipWorkspace({ projectId }: { projectId: string }) {
   880→  const qc = useQueryClient();
   881→  // First try to load saved analysis from DB
   882→  const { data: savedRel } = useQuery({
   883→    queryKey: ["saved-relationships", projectId],
   884→    queryFn: () => api.getSavedRelationships(projectId),
   885→    enabled: !!projectId,
   886→    staleTime: Infinity,
   887→  });
   888→  const { data: freshRel, isLoading: loading, error: relError } = useQuery({
   889→    queryKey: ["source-relationships", projectId],
   890→    queryFn: async () => {
   891→      const res = await fetch("/api/ai/source-relationships", {
   892→        method: "POST",
   893→        headers: { "Content-Type": "application/json" },
   894→        body: JSON.stringify({ projectId }),
   895→      });
   896→      if (!res.ok) {
   897→        const text = await res.text();
   898→        throw new Error(text.slice(0, 200) || `Analysis failed (${res.status})`);
   899→      }
   900→      const data = await res.json();
   901→      qc.invalidateQueries({ queryKey: ["saved-relationships", projectId] });
   902→      return data;
   903→    },
   904→    enabled: false, // Only on manual trigger
   905→  });
   906→  const relMut = useMutation({
   907→    mutationFn: () =>
   908→      fetch("/api/ai/source-relationships", {
   909→        method: "POST",
   910→        headers: { "Content-Type": "application/json" },
   911→        body: JSON.stringify({ projectId }),
   912→      }).then(async (res) => {
   913→        if (!res.ok) {
   914→          const text = await res.text();
   915→          throw new Error(text.slice(0, 200) || `Analysis failed (${res.status})`);
   916→        }
   917→        return res.json();

   920→      qc.invalidateQueries({ queryKey: ["saved-relationships", projectId] });

   942→            </Button>
   943→          </div>
   944→        )}
   945→        {!relMut.isPending && !relError && relData?.skipped && (
   946→          <div className="text-center py-12">
   947→            <Network className="h-10 w-10 mx-auto opacity-40 mb-3" />
